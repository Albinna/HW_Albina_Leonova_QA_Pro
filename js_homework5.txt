// Створюємо об'єкт services
var services = {
    "стрижка": "60 грн",
    "гоління": "80 грн",
    "Миття голови": "100 грн"
};

// Додаємо нову послугу
services['Розбити скло'] = "200 грн";

// Метод для обчислення загальної вартості
services.price = function() {
    let total = 0;

    // Проходимо по всіх значеннях об'єкта
    for (let price of Object.values(this)) {
        // Перевіряємо, чи є значення рядком
        if (typeof price === 'string') {
            // Видаляємо " грн" і перетворюємо рядок у число
            const priceNumber = parseFloat(price.trim().replace(' грн', ''));
            total += priceNumber;
        }
    }

    return total;
};

// Функція для обчислення мінімальної ціни
function minPrice(services) {
    let prices = [];

    // Проходимо по всіх значеннях об'єкта
    for (let price of Object.values(services)) {
        if (typeof price === 'string') {
            const priceNumber = parseFloat(price.trim().replace(' грн', ''));
            prices.push(priceNumber);
        }
    }

    return Math.min(...prices);
}

function maxPrice(services) {
    let prices = [];

    // Проходимо по всіх значеннях об'єкта
    for (let price of Object.values(services)) {
        if (typeof price === 'string') {
            const priceNumber = parseFloat(price.trim().replace(' грн', ''));
            prices.push(priceNumber);
        }
    }

    return Math.max(...prices);
}

// Викликаємо методи і виводимо результат
console.log(services.price()); // Виведе: 440
console.log(minPrice(services)); // Виведе: 60
console.log(maxPrice(services)); // Виведе: 200
