        partners: [
          {

            type: 'subSubCompany',

  function findValueByKey(companyName, clients) {

    for (let client of clients) {
      if (client.name === companyName) {
        return client; 
      }
      
      if (client.partners) {
        const result = findValueByKey(companyName, client.partners); 
        if (result) {
          return result;
        }
      }
    }
    return null; // Якщо компанія не знайдена
  }
